power_rand <- cbc_power(
    nbreaks = 20,
    n_q     = 6,
    data    = data_rand,
    pars    = c('veh_mileage', 'veh_price','battery_condition', 'battery_refurbish', 'battery_range_year0','battery_degradation','charge_freq','charge_DCFC'),
    outcome = "choice",
    obsID   = "obsID"
)
